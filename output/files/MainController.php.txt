<?php
namespace KBK;

use Silex\Application;
use Symfony\Component\HttpFoundation\Request;

class MainController
{
    /**
     * @param Request $request
     * @param Application $app
     * @return mixed
     */
    //action for route:     /
    public function indexAction(Request $request, Application $app)
    {
        // build args array
        $argsArray = array(
            'title' => "Kevin's Vegeburger Kitchen",
            'content' => 'Hello, I am the home page of this site'
        );

        //render template
        $templateName = 'index';
        return $app['twig']->render($templateName . '.html.twig', $argsArray);
    }

    /**
     * @param Request $request
     * @param Application $app
     * @return mixed
     */

        // build arrgs array
        $argsArray = array(
            'title' => 'About us',
            'content'=> 'About us - we`re a great food company'
        );

        //render template
        $templateName = 'about';
        return $app['twig']->render($templateName . '.html.twig', $argsArray);
    }

    /**
     * @param Request $request
     * @param Application $app
     * @return mixed
     */
    //action for route:     /menu
    public function menuAction(Request $request, Application $app)
    {
        // build arrgs array
        $menu = array(
            'Burgers',
            'Salads',
            'Drinks',
            'Deserts'
        );

        $argsArray = array(
            'title' => 'Menu Categories',
            'content'=> $menu
        );

        //render template
        $templateName = 'menu';
        return $app['twig']->render($templateName . '.html.twig', $argsArray);
    }

    /**
     * @param Request $request
     * @param Application $app
     * @return mixed
     */
    //action for route:     /contact
    public function contactAction(Request $request, Application $app)
    {

        $argsArray = array(
            'title' => 'Contact Details',
            'content'=> 'These are our contact details: ...'
        );

        //render template
        $templateName = 'contact';
        return $app['twig']->render($templateName . '.html.twig', $argsArray);
    }
}
